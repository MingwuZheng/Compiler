int prime[3000];

int judge(int ary, int i, int n){
	if(ary == -1){
		if(i * i < n){
			return (1);
		}
	}
	return (0);
}

void Sieve(int n)           //筛法，打质数表
{
	int i, multi, index;
	i = 2;
	prime[0] = -1;
	prime[1] = -1;       //非质数标记为-1,质数标记为0
	if(i * i < n){
		do{
			multi = 2;
			for (index = i * multi; index <= n; multi=multi+1) {
				prime[index] = -1;
				index = i * multi;
			}
			i=i+1;
			if(judge(prime[i], i, n)){
				do {
					i = i + 1;
				}while(judge(prime[i], i, n));
			}
		}while (i * i < n);
	}
}
void main(){
	int n,i;
	scanf(n);
	Sieve(n);
	for(i=0;i<=n;i=i+1){
		if(prime[i]==0)
		printf(" ",i);
	}
}
